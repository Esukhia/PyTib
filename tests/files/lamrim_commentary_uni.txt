དེ་ལྟར་ རྙེད་པ འི་ ཚེ་ སངས་རྒྱས་ ཀྱི་ བཀའ་ ཐམས་ཅད་ དངོས་ དང་ བརྒྱུད་པ འི་ སྒོ་ ནས་ འཚང་རྒྱ་བ འི་ ཐབས་ སྟོན་ བར་ གོ་བ ས། འགའ་ཞིག་ འཚང་རྒྱ་བ འི་ ཐབས་ སྟོན་པ ར་ བཟུང་ ནས་ སློབ་ དགོས་པ་ དང་ ། འགའ་ཞིག་ འཚང་རྒྱ་བ འི་ གེགས་ སུ་ བཟུང་ ནས་ དོར་ བྱར་ བྱེད་པ་ ཆོས་ སྤོང་ གི་ ལས་ ཡིན་པ ས་ གདམས་པ་ འདི་ ལ་ བརྟེན་ ནས་ སྔར་ བཤད་པ་ ལྟར་ བསྟན་པ་ ཐམས་ཅད་ འགལ་ མེད་ དུ་ རྟོགས་པ འི་ ངེས་པ་ རྙེད་པ་ ཙམ་ གྱིས་ དེ་ལྟ་བུ མཁས་པ འི་ སྐྱེ་བོ་ སྐལ་ལྡན་ དུ་མ ས་ བསྟེན་ ཅིང་ ཐུགས་ ཉམས་ སུ་ བཞེས་པ འི་ གདམས་པ་ མཆོག་ ཏུ་ གྱུར་ བ་ སྐྱེས་བུ་གསུམ་ གྱི་ ལམ་ གྱི་ རིམ་པ་ འདི ས་ ཡིད་ རང་དབང་མེད་པ ར་ ཕྲོག་ འཕྲོག་ སྟེ་ ཉན་ བསམ་ སྒོམ་པ་ ལ་ འཇུག་ རིགས་ ཏེ། གདམས་པ་ དེ་ཉིད་ གོང་ དུ་ བཤད་པ་ ལྟར་ གྱི་ ཆེ་བ་ བཞི་ དང་ལྡན་པ ས་ སོ། །རྗེ་བླ་མ ས་ ཀྱང་ ། བསྟན་པ་ ཐམས་ཅད་ འགལ་ མེད་རྟོགས་ པ་ དང་ ། །གསུང་རབ་ མ་ ལུས་ གདམས་པ ར་ འཆར་བ་ དང་ ། །རྒྱལ་བ འི་ དགོངས་པ་ བདེ་བླག་ རྙེད་པ་ དང་ ། །ཉེས་སྤྱོད་ ཆེན་པོ འི་ གཡང་ས་ ལས་ ཀྱང་ སྲུང་ ། །དེ་ཕྱིར་ རྒྱ་བོད་ མཁས་པ འི་ སྐྱེ་བོ་ ནི། །སྐལ་ལྡན་ དུ་མ ས་ བསྟེན་པ འི་ གདམས་པ་ མཆོག་ །སྐྱེས་བུ་གསུམ་ གྱི་ ལམ་ གྱི་ ཇི་ལྟར་ འབད་ སྙམ་ ན་ གོང་ དུ་ བཤད་པ་ ལྟར་ ངན་སོང་ གི་ སྡུག་བསྔལ་ གྱིས་ འཇིགས་པ་ དང་ ། འཇིགས་པ་ དེ་ལས་ སྐྱོབ་པ འི་ ནུས་པ་ དཀོན་མཆོག་གསུམ་ ལ་ ཡོད་པ ར་ ཤེས་ ནས་ དཀོན་མཆོག་ རིན་པོ་ཆེ་ རྣམ་པ་ གསུམ་ ལ་ སྙིང་ཐག་ པ་ ནས་ སྐྱབས་ སུ་ འགྲོ་བ ར་ བྱ་ ལ། དཀོན་མཆོག་གསུམ་ གྱིས་ ངན་སོང་ གི་ འཇིགས་པ་ ལས་ སྐྱོབ་ ནུས་པ་ ཡང་ ཡིན་ ཏེ། སངས་རྒྱས་དཀོན་མཆོག་ དེ་རང་ འཇིགས་པ་ ཐམས་ཅད་ ལས་ གྲོལ། །གཞན་ འཇིགས་པ་ ཐམས་ཅད་ ལས་ སྐྱོབ་པ འི་ ཐབས་ ལ་ མཁས། ཀུན་ ལ་ ཐུགས་རྗེ་ཆེན་པོ་ ཉེ་རིང་མེད་པ ར་ འཇུག་ །ཕན་ བཏགས་མ་ བཏགས་ ཐམས་ཅད་ ཀྱི་ དོན་ མཛད་པ ས་ སྐྱབས་ སུ་ འོས་ ཤིང་ ། དེ ས་ ཆོས་ དང་ དགེ་འདུན་ ཡང་ མཚན་ ནུས་ ལ། །དེས་ན་ ཕྱི་རོལ་ གཉིས་ རྣམས་ ཀྱི་ མཆོག་ སངས་རྒྱས་ ལ་ སྐྱབས་ སུ་ མཆི འོ། །བདག་ འཁོར་བ་ དང་ ངན་སོང་ གི་ སྡུག་བསྔལ་ ལས་ སྒྲོལ་བ འི་ སྟོན་པ་ མཛད་ དུ་ གསོལ། འདོད་ཆགས་ དང་ བྲལ་བ་ རྣམས་ ཀྱི་ མཆོག་ ཆོས་ ལ་ སྐྱབས་ སུ་ མཆི འོ། །བདག་ འཁོར་བ་ དང་ ངན་སོང་ གི་ འཇིགས་པ་ ལས་ སྒྲོལ་བ འི་ སྐྱབས་དངོས་ མཛད་ དུ་ གསོལ། ཚོགས་ རྣམས་ ཀྱི་ མཆོག་ དགེ་འདུན་ ལ་ སྐྱབས་ སུ་ མཆི འོ། །བདག་ འཁོར་བ་ དང་ ངན་སོང་ གི་ སྡུག་བསྔལ་ ལས་ སྒྲོལ་བ འི་ གྲོགས་
